32/// The maximum number of block hashes to keep in the history.
33///
34/// Note: This might be made configurable in the future.
35pub const BLOCK_HASH_COUNT: u32 = 256;
36
37// Accumulates the receipt's events (logs) for the current transaction
38// that are needed to construct the final transaction receipt.
39environmental!(receipt: AccumulateReceipt);
40
41/// Capture the Ethereum log for the current transaction.
42///
43/// This method does nothing if called from outside of the ethereum context.
44pub fn capture_ethereum_log(contract: &H160, data: &[u8], topics: &[H256]) {
45	receipt::with(|receipt| {
46		receipt.add_log(contract, data, topics);
47	});
48}
49
50/// Get the receipt details of the current transaction.
51///
52/// This method returns `None` if and only if the function is called
53/// from outside of the ethereum context.
54pub fn get_receipt_details() -> Option<(Vec<u8>, LogsBloom)> {
55	receipt::with(|receipt| {
56		let encoding = core::mem::take(&mut receipt.encoding);
57		let bloom = core::mem::take(&mut receipt.bloom);
58		(encoding, bloom)
59	})
60}

62/// Capture the receipt events emitted from the current ethereum
63/// transaction. The transaction must be signed by an eth-compatible
64/// wallet.
65pub fn with_ethereum_context<R>(f: impl FnOnce() -> R) -> R {
66	receipt::using(&mut AccumulateReceipt::new(), f)
67}

69/// Clear the storage used to capture the block hash related data.
70pub fn on_initialize<T: Config>() {

72	EthereumBlock::<T>::kill();
73}

75/// Build the ethereum block and store it into the pallet storage.
76pub fn on_finalize_build_eth_block<T: Config>(
77	block_author: H160,
78	eth_block_num: U256,
79	eth_block_base_fee: U256,
80	gas_limit: U256,
81	timestamp: U256,
82) {
83	let parent_hash = if eth_block_num > U256::zero() {
84		BlockHash::<T>::get(eth_block_num - 1)
85	} else {
86		H256::default()
87	};
88
89	let block_builder_ir = EthBlockBuilderIR::<T>::get();
90	EthBlockBuilderIR::<T>::kill();
91
92	// Load the first values if not already loaded.
93	let (block, receipt_data) = EthereumBlockBuilder::<T>::from_ir(block_builder_ir).build(
94		eth_block_num,
95		eth_block_base_fee,
96		parent_hash,
97		timestamp,
98		block_author,
99		gas_limit,
100	);
101
102	// Put the block hash into storage.
103	BlockHash::<T>::insert(eth_block_num, block.hash);
104
105	// Prune older block hashes.
106	let block_hash_count = BLOCK_HASH_COUNT;
107	let to_remove =
108		eth_block_num.saturating_sub(block_hash_count.into()).saturating_sub(One::one());
109	if !to_remove.is_zero() {
110		<BlockHash<T>>::remove(U256::from(UniqueSaturatedInto::<u32>::unique_saturated_into(
111			to_remove,
112		)));
113	}
114	// Store the ETH block into the last block.
115	EthereumBlock::<T>::put(block);
116	// Store the receipt info data for offchain reconstruction.
117	ReceiptInfoData::<T>::put(receipt_data);
118}

120/// Process a transaction payload with extra details.
121/// This stores the RLP encoded transaction and receipt details into storage.
122///
123/// The data is used during the `on_finalize` hook to reconstruct the ETH block.
124pub fn process_transaction<T: Config>(
125	transaction_encoded: Vec<u8>,
126	success: bool,
127	gas_used: Weight,
128) {
129	// Method returns `None` only when called from outside of the ethereum context.
130	// This is not the case here, since this is called from within the
131	// ethereum context.
132	let (encoded_logs, bloom) = get_receipt_details().unwrap_or_default();
133
134	let block_builder_ir = EthBlockBuilderIR::<T>::get();
135	let mut block_builder = EthereumBlockBuilder::<T>::from_ir(block_builder_ir);
136
137	block_builder.process_transaction(transaction_encoded, success, gas_used, encoded_logs, bloom);
138
139	EthBlockBuilderIR::<T>::put(block_builder.to_ir());
140}

142// The `EthereumBlockBuilder` builds the Ethereum-compatible block by maintaining
143// two incremental hash builders. Each builder accumulates entries until the trie
144// is finalized:
145//  1. `transactions_root` - builds the Merkle root of transaction payloads
146//  2. `receipts_root` - builds the Merkle root of transaction receipts (event logs)
147//
148// The `EthereumBlockBuilder` is serialized and deserialized to and from storage
149// on every transaction via the `EthereumBlockBuilderIR` object. This is needed until
150// the runtime exposes a better API to preserve the state between transactions (ie,
151// the global `environment!` is wiped because each transaction will instantiate a new
152// WASM instance).
153//
154// For this reason, we need to account for the memory used by the `EthereumBlockBuilder`
155// and for the pallet storage consumed by the `EthereumBlockBuilderIR`.
156//
157// ## Memory Usage Analysis
158//
159// The incremental hash builder accumulates entries until the trie is finalized.
160// The last added entry value is kept in memory until it can be hashed.
161// The keys are always ordered and the hashing happens when the next entry is added to
162// the trie. The common prefix of the current and previous keys forms the path into the
163// trie, and together with the value of the previous entry, a hash of 32 bytes is
164// computed.
165//
166// For this reason, the memory usage of the incremental hash builder is no greater
167// than two entries of maximum size, plus some marginal book-keeping overhead
168// (ignored to simplify calculations).
169//
170// `IncrementalHashBuilder = 2 * maximum size of the entry`
171//
172// Additionally, the block builder caches the first entry for each incremental hash.
173// The entry is loaded from storage into RAM when either:
174// - The block is finalized, OR
175// - After 127 transactions.
176// Therefore, an additional entry of maximum size is needed in memory.
177//
178// That gives us 3 items of maximum size per each hash builder.
179//
180// `EthereumBlockBuilder = 3 * (max size of transactions + max size of receipts)`
181// The maximum size of a transaction is limited by
182// `limits::MAX_TRANSACTION_PAYLOAD_SIZE`, while the maximum size of a receipt is
183// limited by `limits::EVENT_BYTES`.
184//
185// Similarly, this is the amount of pallet storage consumed by the
186// `EthereumBlockBuilderIR` object, plus a marginal book-keeping overhead.
187pub fn block_builder_bytes_usage(max_events_size: u32) -> u32 {
188	// A block builder requires 3 times the maximum size of the entry.
189	const MEMORY_COEFFICIENT: u32 = 3;
190
191	// Because events are not capped, and the builder cannot exceed the
192	// number of bytes received, the actual memory usage for receipts is:
193	// `receipts_hash_builder = min(events_per_tx * 3, max_events_size)`
194	// where `max_events_size` can be consumed by a single transaction.
195	// Since we don't know in advance the `events_per_tx`, we'll assume the
196	// worst case scenario.
197	let receipts_hash_builder = max_events_size;
198
199	// `transactions_root` hash builder
200	let transactions_hash_builder =
201		limits::MAX_TRANSACTION_PAYLOAD_SIZE.saturating_mul(MEMORY_COEFFICIENT);
202
203	receipts_hash_builder.saturating_add(transactions_hash_builder)
204}
