40/// Ethereum block builder designed to incrementally build the transaction and receipt trie roots.
41///
42/// This builder is optimized to minimize memory usage and pallet storage by leveraging the internal
43/// structure of the Ethereum trie and the RLP encoding of receipts.
44#[derive(DefaultNoBound)]
45pub struct EthereumBlockBuilder<T> {
46	pub(crate) transaction_root_builder: IncrementalHashBuilder,
47	pub(crate) receipts_root_builder: IncrementalHashBuilder,
48	pub(crate) tx_hashes: Vec<H256>,
49	gas_used: U256,
50	logs_bloom: LogsBloom,
51	gas_info: Vec<ReceiptGasInfo>,
52	_phantom: core::marker::PhantomData<T>,
53}
54
55impl<T: crate::Config> EthereumBlockBuilder<T> {
56	/// Converts the builder into an intermediate representation.
57	///
58	/// The intermediate representation is extracted from the pallet storage.
59	pub fn to_ir(self) -> EthereumBlockBuilderIR {
60		EthereumBlockBuilderIR {
61			transaction_root_builder: self.transaction_root_builder.to_ir(),
62			receipts_root_builder: self.receipts_root_builder.to_ir(),
63			gas_used: self.gas_used,
64			tx_hashes: self.tx_hashes,
65			logs_bloom: self.logs_bloom.bloom,
66			gas_info: self.gas_info,
67		}
68	}
69
70	/// Converts the intermediate representation back into a builder.
71	///
72	/// The intermediate representation is placed into the pallet storage.
73	pub fn from_ir(ir: EthereumBlockBuilderIR) -> Self {
74		Self {
75			transaction_root_builder: IncrementalHashBuilder::from_ir(ir.transaction_root_builder),
76			receipts_root_builder: IncrementalHashBuilder::from_ir(ir.receipts_root_builder),
77			gas_used: ir.gas_used,
78			tx_hashes: ir.tx_hashes,
79			logs_bloom: LogsBloom { bloom: ir.logs_bloom },
80			gas_info: ir.gas_info,
81			_phantom: core::marker::PhantomData,
82		}
83	}

85	/// Store the first transaction and receipt in pallet storage.
86	fn pallet_put_first_values(&mut self, values: (Vec<u8>, Vec<u8>)) {
87		crate::EthBlockBuilderFirstValues::<T>::put(Some(values));
88	}
89
90	/// Take the first transaction and receipt from pallet storage.
91	fn pallet_take_first_values(&mut self) -> Option<(Vec<u8>, Vec<u8>)> {
92		crate::EthBlockBuilderFirstValues::<T>::take()
93	}

95	/// Process a single transaction at a time.
96	pub fn process_transaction(
97		&mut self,
98		transaction_encoded: Vec<u8>,
99		success: bool,
100		gas_used: Weight,
101		encoded_logs: Vec<u8>,
102		receipt_bloom: LogsBloom,
103	) {
104		let tx_hash = H256(keccak_256(&transaction_encoded));
105		self.tx_hashes.push(tx_hash);
106
107		// Update the transaction trie.
108		let transaction_type = Self::extract_transaction_type(transaction_encoded.as_slice());
109
110		// Update gas and logs bloom.
111		self.gas_used = self.gas_used.saturating_add(gas_used.ref_time().into());
112		self.logs_bloom.accrue_bloom(&receipt_bloom);
113
114		// Update the receipt trie.
115		let encoded_receipt = AccumulateReceipt::encoded_receipt(
116			encoded_logs,
117			receipt_bloom,
118			success,
119			self.gas_used.as_u64(),
120			transaction_type,
121		);
122
123		self.gas_info.push(ReceiptGasInfo { gas_used: gas_used.ref_time().into() });
124
125		// The first transaction and receipt are returned to be stored in the pallet storage.
126		// The index of the incremental hash builders already expects the next items.
127		if self.tx_hashes.len() == 1 {
128			log::debug!(target: LOG_TARGET, "Storing first transaction and receipt in pallet storage");
129			self.pallet_put_first_values((transaction_encoded, encoded_receipt));
130			return;
131		}
132
133		if self.transaction_root_builder.needs_first_value(BuilderPhase::ProcessingValue) {
134			if let Some((first_tx, first_receipt)) = self.pallet_take_first_values() {
135				log::debug!(target: LOG_TARGET, "Loaded first transaction and receipt from pallet storage");
136				self.transaction_root_builder.set_first_value(first_tx);
137				self.receipts_root_builder.set_first_value(first_receipt);
138			} else {
139				log::error!(target: LOG_TARGET, "First transaction and receipt must be present at processing phase");
140			}
141		}
142
143		self.transaction_root_builder.add_value(transaction_encoded);
144		self.receipts_root_builder.add_value(encoded_receipt);
145	}

147	/// Build the ethereum block from provided data.
148	pub fn build(
149		&mut self,
150		block_number: U256,
151		base_fee_per_gas: U256,
152		parent_hash: H256,
153		timestamp: U256,
154		block_author: H160,
155		gas_limit: U256,
156	) -> (Block, Vec<ReceiptGasInfo>) {
157		if self.transaction_root_builder.needs_first_value(BuilderPhase::Build) {
158			if let Some((first_tx, first_receipt)) = self.pallet_take_first_values() {
159				self.transaction_root_builder.set_first_value(first_tx);
160				self.receipts_root_builder.set_first_value(first_receipt);
161			} else {
162				log::debug!(target: LOG_TARGET, "Building an empty block");
163			}
164		}
165
166		let transactions_root = self.transaction_root_builder.finish();
167		let receipts_root = self.receipts_root_builder.finish();
168
169		let tx_hashes = core::mem::replace(&mut self.tx_hashes, Vec::new());
170		let gas_info = core::mem::replace(&mut self.gas_info, Vec::new());
171
172		let mut block = Block {
173			number: block_number,
174			parent_hash,
175			timestamp,
176			miner: block_author,
177			gas_limit,
178
179			state_root: transactions_root,
180			transactions_root,
181			receipts_root,
182
183			base_fee_per_gas,
184			gas_used: self.gas_used,
185
186			logs_bloom: self.logs_bloom.bloom.into(),
187			transactions: HashesOrTransactionInfos::Hashes(tx_hashes),
188
189			..Default::default()
190		};
191
192		let block_hash = block.header_hash();
193		block.hash = block_hash;
194
195		(block, gas_info)
196	}
197
198	/// Extracts the transaction type from the RLP encoded transaction.
199	///
200	/// This is needed to build the RLP encoding of the receipt.
201	fn extract_transaction_type(transaction_encoded: &[u8]) -> Vec<u8> {
202		// The transaction type is the first byte from the encoded transaction,
203		// when the transaction is not legacy. For legacy transactions, there's
204		// no type defined. Additionally, the RLP encoding of the tx type byte
205		// is identical to the tx type.
206		transaction_encoded
207			.first()
208			.cloned()
209			.map(|first| match first {
210				TYPE_EIP2930 | TYPE_EIP1559 | TYPE_EIP4844 | TYPE_EIP7702 => vec![first],
211				_ => vec![],
212			})
213			.unwrap_or_default()
214	}
215}

217/// The intermediate representation of the Ethereum block builder.
218#[derive(Encode, Decode, TypeInfo)]
219pub struct EthereumBlockBuilderIR {
220	transaction_root_builder: IncrementalHashBuilderIR,
221	receipts_root_builder: IncrementalHashBuilderIR,
222	gas_used: U256,
223	logs_bloom: [u8; BLOOM_SIZE_BYTES],
224	pub(crate) tx_hashes: Vec<H256>,
225	pub(crate) gas_info: Vec<ReceiptGasInfo>,
226}
227
228impl Default for EthereumBlockBuilderIR {
229	fn default() -> Self {
230		Self {
231			// Default not implemented for [u8; BLOOM_SIZE_BYTES]
232			logs_bloom: [0; BLOOM_SIZE_BYTES],
233			transaction_root_builder: IncrementalHashBuilderIR::default(),
234			receipts_root_builder: IncrementalHashBuilderIR::default(),
235			gas_used: U256::zero(),
236			tx_hashes: Vec::new(),
237			gas_info: Vec::new(),
238		}
239	}

252	/// Manual implementation of the Ethereum trie root computation.
253	///
254	/// Given the RLP encoded values, the implementation adjusts the
255	/// index to account for RLP encoding rules.
256	fn manual_trie_root_compute(encoded: Vec<Vec<u8>>) -> H256 {
257		const fn adjust_index_for_rlp(i: usize, len: usize) -> usize {
258			if i > 0x7f {
259				i
260			} else if i == 0x7f || i + 1 == len {
261				0
262			} else {
263				i + 1
264			}
265		}
266
267		let mut hb = HashBuilder::default();
268
269		let items_len = encoded.len();
270		for i in 0..items_len {
271			let index = adjust_index_for_rlp(i, items_len);
272
273			let index_buffer = rlp::encode_fixed_size(&index);
274			hb.add_leaf(Nibbles::unpack(&index_buffer), &encoded[index]);
275
276			// Each mask in these vectors holds a u16.
277			let masks_len = (hb.state_masks.len() + hb.tree_masks.len() + hb.hash_masks.len()) * 2;
278			let _size = hb.key.len() +
279				hb.value.as_slice().len() +
280				hb.stack.len() * 33 +
281				masks_len + hb.rlp_buf.len();
282		}
283
284		hb.root().0.into()
285	}

287	/// The test compares three hashing options:
288	/// - Block::compute_trie_root: this uses the consensus proofs crate
289	/// - manual_trie_root_compute: this ensures the keys are added in the correct order
290	/// - IncrementalHashBuilder: this offers the most compact storage option
291	///
292	/// The above hashes must be identical. While at it, the incremental hash
293	/// builder is serialized and deserialized to ensure consistency.
294	#[test]
295	fn incremental_hasher() {
296		const UPPER_BOUND: usize = 256;
297		const RLP_VALUE_SIZE: usize = 128;
298
299		let mut rlp_values = Vec::with_capacity(UPPER_BOUND);
300
301		for i in 0..UPPER_BOUND {
302			// Simulate an RLP value repeated for `i`.
303			let rlp_value = vec![i as u8; RLP_VALUE_SIZE];
304
305			rlp_values.push(rlp_value);
306
307			let block_hash: H256 = Block::compute_trie_root(&rlp_values).0.into();
308			let manual_hash = manual_trie_root_compute(rlp_values.clone());
309
310			let mut first_value = Some(rlp_values[0].clone());
311			let mut builder = IncrementalHashBuilder::default();
312			for rlp_value in rlp_values.iter().skip(1) {
313				if builder.needs_first_value(BuilderPhase::ProcessingValue) {
314					let value = first_value.take().expect("First value must be present; qed");
315					builder.set_first_value(value);
316				}
317				builder.add_value(rlp_value.clone());
318
319				let ir_builder = builder.to_ir();
320				builder = IncrementalHashBuilder::from_ir(ir_builder);
321			}
322			if let Some(value) = first_value.take() {
323				builder.set_first_value(value);
324			}
325			let incremental_hash = builder.finish();
326
327			assert_eq!(block_hash, manual_hash);
328			assert_eq!(block_hash, incremental_hash);
329		}
330	}

332	#[test]
333	fn test_alloy_rlp_ordering_compatibility() {
334		let zero_encoded = rlp::encode_fixed_size(&0usize);
335		let max_single_byte = rlp::encode_fixed_size(&127usize);
336		let first_multi_byte = rlp::encode_fixed_size(&128usize);
337
338		// Document the exact bytes we expect
339		assert_eq!(zero_encoded.as_slice(), &[0x80]); // RLP encoding of 0
340		assert_eq!(max_single_byte.as_slice(), &[0x7f]); // RLP encoding of 127
341		assert_eq!(first_multi_byte.as_slice(), &[0x81, 0x80]); // RLP encoding of 128
342
343		// Verify ordering
344		assert!(max_single_byte < zero_encoded);
345		assert!(zero_encoded < first_multi_byte);
346	}

348	#[test]
349	fn ensure_identical_hashes() {
350		// curl -X POST --data '{"jsonrpc":"2.0","method":"eth_getBlockByNumber","params":["0x161bd0f", true],"id":1}' -H "Content-Type: application/json" https://ethereum-rpc.publicnode.com | jq .result
351		const BLOCK_PATH: &str = "./test-assets/eth_block.json";
352		// curl -X POST --data '{"jsonrpc":"2.0","method":"eth_getBlockReceipts","params":["0x161bd0f"],"id":1}' -H "Content-Type: application/json" https://ethereum-rpc.publicnode.com | jq .result
353		const BLOCK_RECEIPTS: &str = "./test-assets/eth_receipts.json";
354
355		let json = std::fs::read_to_string(BLOCK_PATH).unwrap();
356		let block: Block = serde_json::from_str(&json).unwrap();
357
358		let json = std::fs::read_to_string(BLOCK_RECEIPTS).unwrap();
359		let receipts: Vec<ReceiptInfo> = serde_json::from_str(&json).unwrap();
360
361		assert_eq!(block.header_hash(), receipts[0].block_hash);
362
363		let tx = match &block.transactions {
364			HashesOrTransactionInfos::TransactionInfos(infos) => infos.clone(),
365			_ => panic!("Expected full tx body"),
366		};
367
368		let encoded_tx: Vec<_> = tx
369			.clone()
370			.into_iter()
371			.map(|tx| tx.transaction_signed.signed_payload())
372			.collect();
373
374		let transaction_details: Vec<_> = tx
375			.into_iter()
376			.zip(receipts.into_iter())
377			.map(|(tx_info, receipt_info)| {
378				if tx_info.transaction_index != receipt_info.transaction_index {
379					panic!("Transaction and receipt index do not match");
380				}
381
382				let logs: Vec<_> = receipt_info
383					.logs
384					.into_iter()
385					.map(|log| (log.address, log.data.unwrap_or_default().0, log.topics))
386					.collect();
387
388				(
389					tx_info.transaction_signed.signed_payload(),
390					logs,
391					receipt_info.status.unwrap_or_default() == 1.into(),
392					receipt_info.gas_used.as_u64(),
393				)
394			})
395			.collect();
396
397		ExtBuilder::default().build().execute_with(|| {
398			// Build the ethereum block incrementally.
399			let mut incremental_block = EthereumBlockBuilder::<Test>::default();
400			for (signed, logs, success, gas_used) in transaction_details {
401				let mut log_size = 0;
402
403				let mut accumulate_receipt = AccumulateReceipt::new();
404				for (address, data, topics) in &logs {
405					let current_size = data.len() + topics.len() * 32 + 20;
406					log_size += current_size;
407					accumulate_receipt.add_log(address, data, topics);
408				}
409
410				incremental_block.process_transaction(
411					signed,
412					success,
413					gas_used.into(),
414					accumulate_receipt.encoding,
415					accumulate_receipt.bloom,
416				);
417
418				let ir = incremental_block.to_ir();
419				incremental_block = EthereumBlockBuilder::from_ir(ir);
420				println!(" Log size {:?}", log_size);
421			}
422
423			// The block hash would differ here because we don't take into account
424			// the ommers and other fields from the substrate perspective.
425			// However, the state roots must be identical.
426			let built_block = incremental_block
427				.build(
428					block.number,
429					block.base_fee_per_gas,
430					block.parent_hash,
431					block.timestamp,
432					block.miner,
433					Default::default(),
434				)
435				.0;
436
437			assert_eq!(built_block.gas_used, block.gas_used);
438			assert_eq!(built_block.logs_bloom, block.logs_bloom);
439			// We are using the tx root for state root.
440			assert_eq!(built_block.state_root, built_block.transactions_root);
441
442			// Double check the receipts roots.
443			assert_eq!(built_block.receipts_root, block.receipts_root);
444
445			let manual_hash = manual_trie_root_compute(encoded_tx.clone());
446
447			let mut total_size = 0;
448			for enc in &encoded_tx {
449				total_size += enc.len();
450			}
451			println!("Total size used by transactions: {:?}", total_size);
452
453			let mut builder = IncrementalHashBuilder::default();
454			let mut loaded = false;
455			for tx in encoded_tx.iter().skip(1) {
456				if builder.needs_first_value(BuilderPhase::ProcessingValue) {
457					loaded = true;
458					let first_tx = encoded_tx[0].clone();
459					builder.set_first_value(first_tx);
460				}
461				builder.add_value(tx.clone())
462			}
463			if !loaded {
464				// Not loaded, therefore the first value must be set now.
465				assert!(builder.needs_first_value(BuilderPhase::Build));
466
467				let first_tx = encoded_tx[0].clone();
468				builder.set_first_value(first_tx);
469			}
470
471			let incremental_hash = builder.finish();
472
473			println!("Incremental hash: {:?}", incremental_hash);
474			println!("Manual Hash: {:?}", manual_hash);
475			println!("Built block Hash: {:?}", built_block.transactions_root);
476			println!("Real Block Tx Hash: {:?}", block.transactions_root);
477
478			assert_eq!(incremental_hash, block.transactions_root);
479
480			// This double checks the compute logic.
481			assert_eq!(manual_hash, block.transactions_root);
482			// This ensures we can compute the same transaction root as Ethereum.
483			assert_eq!(block.transactions_root, built_block.transactions_root);
484		});
485	}
